data => {

    const totals = data.reduce((acc, item) => {
        acc.age+=item.age;
        acc.active += +item.isActive;
        acc.inactive += +!item.isActive;
        acc.eyeColors.counts[item.eyeColor] = acc.eyeColors.counts[item.eyeColor] ? acc.eyeColors.counts[item.eyeColor] + 1 : 1;
        return acc;
        
    }, {
        age: 0,
        active: 0,
        inactive: 0,
        eyeColors:{
            counts:{}
        }
    });

    return {
        averageAge: totals.age/data.length,
        eyeColors:{
            counts: totals.eyeColors.counts,
            totalColors: Object.keys(totals.eyeColors.counts).length
        },
        activeCount: totals.active,
        inactiveCount: totals.inactive,
        totalCount: data.length

    }

}
